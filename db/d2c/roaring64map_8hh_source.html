377  // If the outer map is empty, end_high is less than the first key,
378  // or start_high is greater than the last key, then exit now because
379  // there is no work to do.
380  if (roarings.empty() || end_high < roarings.cbegin()->first ||

385  // If we get here, start_iter points to the first entry in the outer map
386  // with key >= start_high. Such an entry is known to exist (i.e. the
387  // iterator will not be equal to end()) because start_high <= the last
388  // key in the map (thanks to the above if statement).
389  auto start_iter = roarings.lower_bound(start_high);
390  // end_iter points to the first entry in the outer map with
391  // key >= end_high, if such a key exists. Otherwise, it equals end().
392  auto end_iter = roarings.lower_bound(end_high);
393 
394  // Note that the 'lower_bound' method will find the start and end slots,
395  // if they exist; otherwise it will find the next-higher slots.
396  // In the case where 'start' landed on an existing slot, we need to do a
397  // partial erase of that slot, and likewise for 'end'. But all the slots
398  // in between can be fully erased. More precisely:
399  //
400  // 1. If the start point falls on an existing entry, there are two
401  // subcases:
402  // a. if the end point falls on that same entry, remove the closed
403  // interval [start_low, end_low] from that entry and we are done.
404  // b. Otherwise, remove the closed interval [start_low, uint32_max]
405  // from that entry, advance start_iter, and fall through to step 2.
406  // 2. Completely erase all slots in the half-open interval
407  // [start_iter, end_iter)
408  // 3. If the end point falls on an existing entry, remove the closed
409  // interval [0, end_high] from it.
410 
411  // Step 1. If the start point falls on an existing entry...
412  if (start_iter->first == start_high) {
413  auto &start_inner = start_iter->second;
414  // 1a. if the end point falls on that same entry...
415  if (start_iter == end_iter) {
416  start_inner.removeRangeClosed(start_low, end_low);
417  eraseIfEmpty(start_iter);
418  return;
419  }
420 
421  // 1b. Otherwise, remove the closed range [start_low, uint32_max]...
422  start_inner.removeRangeClosed(start_low, uint32_max);

424  // bitmap we just modified for emptiness and erase if it necessary.
425  auto temp = start_iter++;

505  // Logic table summarizing what to do when a given outer key is
506  // present vs. absent from self and other.
507  //
508  // self other (self & other) work to do
509  // --------------------------------------------
510  // absent absent empty None
511  // absent present empty None
512  // present absent empty Erase self
513  // present present empty or not Intersect self with other, but
514  // erase self if result is empty.
515  //
516  // Because there is only work to do when a key is present in 'self', the
517  // main for loop iterates over entries in 'self'.
518 
519  decltype(roarings.begin()) self_next;
520  for (auto self_iter = roarings.begin(); self_iter != roarings.end();
521  self_iter = self_next) {
522  // Do the 'next' operation now, so we don't have to worry about
523  // invalidation of self_iter down below with the 'erase' operation.
524  self_next = std::next(self_iter);
525 
526  auto self_key = self_iter->first;
527  auto &self_bitmap = self_iter->second;
528 
529  auto other_iter = other.roarings.find(self_key);
530  if (other_iter == other.roarings.end()) {
531  // 'other' doesn't have self_key. In the logic table above,
532  // this reflects the case (self.present & other.absent).
533  // So, erase self.
534  roarings.erase(self_iter);
535  continue;
536  }

539  // the case (self.present & other.present). So, intersect self with
540  // other.

542  self_bitmap &= other_bitmap;

918  void flipClosed(uint64_t min, uint64_t max) {
919  if (min > max) {
920  return;
921  }
922  uint32_t start_high = highBytes(min);
923  uint32_t start_low = lowBytes(min);
924  uint32_t end_high = highBytes(max);
925  uint32_t end_low = lowBytes(max);
926 
927  // We put std::numeric_limits<>::max in parentheses to avoid a
928  // clash with the Windows.h header under Windows.
929  const uint32_t uint32_max = (std::numeric_limits<uint32_t>::max)();
930 
931  // Fill in any nonexistent slots with empty Roarings. This simplifies
932  // the logic below, allowing it to simply iterate over the map between
933  // 'start_high' and 'end_high' in a linear fashion.
934  auto current_iter = ensureRangePopulated(start_high, end_high);
935 
936  // If start and end land on the same inner bitmap, then we can do the
937  // whole operation in one call.
938  if (start_high == end_high) {
939  auto &bitmap = current_iter->second;
940  bitmap.flipClosed(start_low, end_low);
941  eraseIfEmpty(current_iter);
942  return;
943  }
944 
945  // Because start and end don't land on the same inner bitmap,
946  // we need to do this in multiple steps:
947  // 1. Partially flip the first bitmap in the closed interval
948  // [start_low, uint32_max]
949  // 2. Flip intermediate bitmaps completely: [0, uint32_max]
950  // 3. Partially flip the last bitmap in the closed interval
951  // [0, end_low]
952 
953  auto num_intermediate_bitmaps = end_high - start_high - 1;
954 
955  // 1. Partially flip the first bitmap.
956  {
957  auto &bitmap = current_iter->second;
958  bitmap.flipClosed(start_low, uint32_max);
959  auto temp = current_iter++;
960  eraseIfEmpty(temp);
961  }
962 
963  // 2. Flip intermediate bitmaps completely.
964  for (uint32_t i = 0; i != num_intermediate_bitmaps; ++i) {
965  auto &bitmap = current_iter->second;
966  bitmap.flipClosed(0, uint32_max);
967  auto temp = current_iter++;
968  eraseIfEmpty(temp);
969  }
970 
971  // 3. Partially flip the last bitmap.
972  auto &bitmap = current_iter->second;
973  bitmap.flipClosed(0, end_low);
974  eraseIfEmpty(current_iter);
975  }

1389  static Roaring64Map fastunion(size_t n, const Roaring64Map **inputs) {
1390  // The strategy here is to basically do a "group by" operation.
1391  // We group the input roarings by key, do a 32-bit
1392  // roaring_bitmap_or_many on each group, and collect the results.
1393  // We accomplish the "group by" operation using a priority queue, which
1394  // tracks the next key for each of our input maps. At each step, our
1395  // algorithm takes the next subset of maps that share the same next key,
1396  // runs roaring_bitmap_or_many on those bitmaps, and then advances the
1397  // current_iter on all the affected entries and then repeats.
1398 
1399  // There is an entry in our priority queue for each of the 'n' inputs.
1400  // For a given Roaring64Map, we look at its underlying 'roarings'
1401  // std::map, and take its begin() and end(). This forms our half-open
1402  // interval [current_iter, end_iter), which we keep in the priority
1403  // queue as a pq_entry. These entries are updated (removed and then
1404  // reinserted with the pq_entry.iterator field advanced by one step) as
1405  // our algorithm progresses. But when a given interval becomes empty
1406  // (i.e. pq_entry.iterator == pq_entry.end) it is not returned to the
1407  // priority queue.
1408  struct pq_entry {
1409  roarings_t::const_iterator iterator;
1410  roarings_t::const_iterator end;
1411  };
1412 
1413  // Custom comparator for the priority queue.
1414  auto pq_comp = [](const pq_entry &lhs, const pq_entry &rhs) {
1415  auto left_key = lhs.iterator->first;
1416  auto right_key = rhs.iterator->first;
1417 
1418  // We compare in the opposite direction than normal because priority
1419  // queues normally order from largest to smallest, but we want
1420  // smallest to largest.
1421  return left_key > right_key;
1422  };
1423 
1424  // Create and populate the priority queue.
1425  std::priority_queue<pq_entry, std::vector<pq_entry>, decltype(pq_comp)> pq(pq_comp);
1426  for (size_t i = 0; i < n; ++i) {
1427  const auto &roarings = inputs[i]->roarings;
1428  if (roarings.begin() != roarings.end()) {
1429  pq.push({roarings.begin(), roarings.end()});
1430  }
1431  }
1432 
1433  // A reusable vector that holds the pointers to the inner bitmaps that
1434  // we pass to the underlying 32-bit fastunion operation.
1435  std::vector<const roaring_bitmap_t*> group_bitmaps;
1436 
1437  // Summary of the algorithm:
1438  // 1. While the priority queue is not empty:
1439  // A. Get its lowest key. Call this group_key
1440  // B. While the lowest entry in the priority queue has a key equal to
1441  // group_key:
1442  // 1. Remove this entry (the pair {current_iter, end_iter}) from
1443  // the priority queue.
1444  // 2. Add the bitmap pointed to by current_iter to a list of
1445  // 32-bit bitmaps to process.
1446  // 3. Advance current_iter. Now it will point to a bitmap entry
1447  // with some key greater than group_key (or it will point to
1448  // end()).
1449  // 4. If current_iter != end_iter, reinsert the pair into the
1450  // priority queue.
1451  // C. Invoke the 32-bit roaring_bitmap_or_many() and add to result
1452  Roaring64Map result;
1453  while (!pq.empty()) {
1454  // Find the next key (the lowest key) in the priority queue.
1455  auto group_key = pq.top().iterator->first;
1456 
1457  // The purpose of the inner loop is to gather all the inner bitmaps
1458  // that share "group_key" into "group_bitmaps" so that they can be
1459  // fed to roaring_bitmap_or_many(). While we are doing this, we
1460  // advance those iterators to their next value and reinsert them
1461  // into the priority queue (unless they reach their end).
1462  group_bitmaps.clear();
1463  while (!pq.empty()) {
1464  auto candidate_current_iter = pq.top().iterator;
1465  auto candidate_end_iter = pq.top().end;
1466 
1467  auto candidate_key = candidate_current_iter->first;
1468  const auto &candidate_bitmap = candidate_current_iter->second;
1469 
1470  // This element will either be in the group (having
1471  // key == group_key) or it will not be in the group (having
1472  // key > group_key). (Note it cannot have key < group_key
1473  // because of the ordered nature of the priority queue itself
1474  // and the ordered nature of all the underlying roaring maps).
1475  if (candidate_key != group_key) {
1476  // This entry, and (thanks to the nature of the priority
1477  // queue) all other entries as well, are all greater than
1478  // group_key, so we're done collecting elements for the
1479  // current group. Because of the way this loop was written,
1480  // the group will will always contain at least one element.
1481  break;
1482  }
1483 
1484  group_bitmaps.push_back(&candidate_bitmap.roaring);
1485  // Remove this entry from the priority queue. Note this
1486  // invalidates pq.top() so make sure you don't have any dangling
1487  // references to it.
1488  pq.pop();
1489 
1490  // Advance 'candidate_current_iter' and insert a new entry
1491  // {candidate_current_iter, candidate_end_iter} into the
1492  // priority queue (unless it has reached its end).
1493  ++candidate_current_iter;
1494  if (candidate_current_iter != candidate_end_iter) {
1495  pq.push({candidate_current_iter, candidate_end_iter});
1496  }
1497  }
1498 
1499  // Use the fast inner union to combine these.
1500  auto *inner_result = roaring_bitmap_or_many(group_bitmaps.size(),
1501  group_bitmaps.data());
1502  // Insert the 32-bit result at end of the 'roarings' map of the
1503  // result we are building.
1504  result.roarings.insert(result.roarings.end(),
1505  std::make_pair(group_key, Roaring(inner_result)));
1506  }
1507  return result;
1508  }
